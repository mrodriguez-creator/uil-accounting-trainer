STEP 3: Add JavaScript functions (paste before closing </script>)
â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

let currentMathProblem = null;
let mathProgress = JSON.parse(localStorage.getItem('mathProgress')) || {
    completed: [], correct: [], points: 0
};

function showMathCategory(category) {
    const problems = mathProblems[category] || [];
    let html = `<div class="hero"><h1>Math Practice</h1></div>`;
    
    problems.forEach(p => {
        const completed = mathProgress.completed.includes(p.id) ? 'âœ“ ' : '';
        html += `
            <div class="problem-card" onclick="showMathProblem('${category}','${p.id}')">
                <span class="badge badge-${p.difficulty}">${p.difficulty}</span>
                <h3>${completed}${p.title}</h3>
                <p>${p.problem}</p>
            </div>
        `;
    });
    
    document.getElementById('mathContent').innerHTML = html;
    showSection('math-viewer');
}

function showMathProblem(category, id) {
    const problem = mathProblems[category].find(p => p.id === id);
    currentMathProblem = problem;
    
    const html = `
        <div class="hero"><h1>${problem.title}</h1></div>
        <div class="lesson-section">
            <h3>Problem:</h3>
            <p style="font-size:1.125rem;">${problem.problem}</p>
            
            <div style="margin:2rem 0;">
                <input type="number" id="mathAnswer" placeholder="Your answer" 
                       style="padding:1rem;font-size:1.25rem;width:200px;
                              background:rgba(15,23,42,0.8);border:2px solid var(--border);
                              border-radius:0.5rem;color:var(--text-light);">
                <button class="btn" onclick="checkMathAnswer()">Check Answer</button>
            </div>
            
            <div id="mathFeedback" style="margin:1rem 0;"></div>
            
            <button class="btn btn-secondary" onclick="showMathSolution()">
                Show Solution
            </button>
            
            <div id="mathSolution" style="display:none;margin-top:2rem;
                 background:rgba(59,130,246,0.1);padding:1.5rem;border-radius:0.5rem;">
                <h3>Solution:</h3>
                <p>${problem.solution}</p>
                <p style="margin-top:1rem;"><strong>Answer: $${problem.answer.toFixed(2)}</strong></p>
            </div>
        </div>
    `;
    
    document.getElementById('mathContent').innerHTML = html;
}

function checkMathAnswer() {
    if (!currentMathProblem) return;
    
    const input = parseFloat(document.getElementById('mathAnswer').value);
    const isCorrect = Math.abs(input - currentMathProblem.answer) < 0.01;
    const feedback = document.getElementById('mathFeedback');
    
    if (isCorrect) {
        feedback.innerHTML = '<div style="color:var(--secondary);font-weight:600;">âœ“ Correct!</div>';
        if (!mathProgress.correct.includes(currentMathProblem.id)) {
            mathProgress.correct.push(currentMathProblem.id);
            mathProgress.points += 10;
        }
        if (!mathProgress.completed.includes(currentMathProblem.id)) {
            mathProgress.completed.push(currentMathProblem.id);
        }
        localStorage.setItem('mathProgress', JSON.stringify(mathProgress));
        updateMathStats();
    } else {
        feedback.innerHTML = '<div style="color:var(--danger);font-weight:600;">âœ— Try again</div>';
    }
}

function showMathSolution() {
    document.getElementById('mathSolution').style.display = 'block';
    if (!mathProgress.completed.includes(currentMathProblem.id)) {
        mathProgress.completed.push(currentMathProblem.id);
        localStorage.setItem('mathProgress', JSON.stringify(mathProgress));
        updateMathStats();
    }
}

function updateMathStats() {
    document.getElementById('mathCompleted').textContent = mathProgress.completed.length;
    const accuracy = mathProgress.completed.length > 0
        ? Math.round((mathProgress.correct.length / mathProgress.completed.length) * 100)
        : 0;
    document.getElementById('mathAccuracy').textContent = accuracy + '%';
}

// Initialize on page load
document.addEventListener('DOMContentLoaded', updateMathStats);
